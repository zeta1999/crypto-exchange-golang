Crypto Exchange Console

This lightweight GUI submits HTTP orders and streams WebSocket snapshots.

fetch('/orders', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'Authorization': 'Bearer dev-secret-token'
  },
  body: JSON.stringify({
    instrument: 'BTC-USD',
    price: 42000,
    volume: 0.5,
    side: 'buy',
    client_id: 'demo-1'
  })
});

const scheme = window.location.protocol === 'https:' ? 'wss' : 'ws';
const ws = new WebSocket(`${scheme}://localhost:8081/ws`);
ws.onopen = () => ws.send(JSON.stringify({
  command: 'snapshot',
  instrument: 'BTC-USD',
  token: 'dev-secret-token'
}));
ws.onmessage = (event) => console.log(JSON.parse(event.data));
